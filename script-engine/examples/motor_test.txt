# Motor positioning test script
# Tests motor movement to multiple positions

PRINT ================================================
PRINT Motor Test Script
PRINT Started at {timestamp}
PRINT ================================================

# Configuration
SET target1 = 0
SET target2 = 45
SET target3 = 90
SET target4 = -45
SET tolerance = 2.0
SET max_time = 5000

# Test position 1
PRINT
PRINT Test 1: Moving to {target1} degrees
SEND DP{target1}
WAIT 500
SENDNW GPS
RECV PS:{position}
PRINT Result: {position} degrees
SET error = ABS(target1 - position)
PRINT Error: {error} degrees
